\doxysubsection{Detailed Description}
\mbox{\hyperlink{classostk_1_1mathematics_1_1geometry_1_1d2_1_1object_1_1_polygon}{Polygon}}. 

\begin{DoxyVerb}                        A plane figure that is bounded by a finite chain of straight line segments closing in a
                        loop to form a closed polygonal chain or circuit. These segments are called its edges,
                        and the points where two edges meet are the polygon's vertices.
\end{DoxyVerb}
