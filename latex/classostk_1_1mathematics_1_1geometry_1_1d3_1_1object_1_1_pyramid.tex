\doxysubsection{Detailed Description}
\mbox{\hyperlink{classostk_1_1mathematics_1_1geometry_1_1d3_1_1object_1_1_pyramid}{Pyramid}}. 

\begin{DoxyVerb}                        A pyramid is a polyhedron formed by connecting a polygonal base and a point, called the
                        apex. Each base edge and apex form a triangle, called a lateral face.
\end{DoxyVerb}
