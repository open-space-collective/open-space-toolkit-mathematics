\doxysubsection{Detailed Description}
\mbox{\hyperlink{classostk_1_1mathematics_1_1curvefitting_1_1interpolator_1_1_cubic_spline}{Cubic\+Spline}}. 

\begin{DoxyVerb}                        A cubic Spline interpolator is a spline where each piece is a third-degree polynomial
                        specified in Spline form, that is by its values and first derivatives at the end points
                        of the corresponding domain interval.
\end{DoxyVerb}
